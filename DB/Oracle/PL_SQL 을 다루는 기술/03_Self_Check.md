           FROM EMPLOYEES e
           WHERE MANAGER_ID=145) employee
    ON (ex.EMPLOYEE_ID = employee.EMPLOYEE_ID)
WHEN MATCHED THEN
  UPDATE SET ex.BONUS_AMT = employee.SALARY * 0.01
WHEN NOT MATCHED THEN
  INSERT (ex.EMPLOYEE_ID, ex.BONUS_AMT)
  VALUES (employee.EMPLOYEE_ID, employee.SALARY * 0.005);
